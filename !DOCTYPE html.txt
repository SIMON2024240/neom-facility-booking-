                updateSectionB(booking);
            }
        }

        // Update Section B display
        function updateSectionB(booking) {

            }

                document.getElementById('approvalDisplay').textContent = '☑ Yes by: ' + booking.approvedBy;
                document.getElementById('approvalDateDisplay').textContent = booking.approvedAt ? new Date(booking.approvedAt).toLocaleDateString() : '-';
            } else if (booking.status === 'rejected') {
                document.getElementById('approvalDisplay').textContent = '☑ No';
                document.getElementById('approvalDateDisplay').textContent = booking.rejectedAt ? new Date(booking.rejectedAt).toLocaleDateString() : '-';
                document.getElementById('adminRemarksDisplay').textContent = booking.rejectionReason || 'Request rejected';
            }
        }

        // Generate reports
        function generateReport(type, format) {
            var data = [];
            var filename = '';
            
            if (type === 'bookings') {
                for (var i = 0; i < bookings.length; i++) {
                    var b = bookings[i];
                    data.push({
                        'Request ID': b.id,
                        'Requester': b.requesterName,
                        'Company': b.companyName,
                        'Facility': b.venueRequested,
                        'Event': b.event,
                        'Start Date': b.startDate,
                        'End Date': b.endDate,
                        'Status': b.status,
                        'Source': b.source
                    });
                }
                filename = 'nrc9_booking_report';
            } else if (type === 'facilities') {
                for (var i = 0; i < facilities.length; i++) {
                    var f = facilities[i];
                    var totalBookings = 0;
                    for (var j = 0; j < bookings.length; j++) {
                        if (bookings[j].venueRequested === f.name) {
                            totalBookings++;
                        }
                    }
                    data.push({
                        'Facility ID': f.id,
                        'Name': f.name,
                        'Capacity': f.capacity,
                        'Description': f.description,
                        'Total Bookings': totalBookings
                    });
                }
                filename = 'nrc9_facility_report';
            } else if (type === 'users') {
                for (var i = 0; i < users.length; i++) {
                    var u = users[i];
                    var handledBookings = 0;
                    for (var j = 0; j < bookings.length; j++) {
                        if (bookings[j].handledBy === u.name) {
                            handledBookings++;
                        }
                    }
                    data.push({
                        'User ID': u.id,
                        'Name': u.name,
                        'Email': u.email,
                        'Role': u.role,
                        'Bookings Handled': handledBookings
                    });
                }
                filename = 'nrc9_user_report';
            }
            
            if (format === 'excel' && data.length > 0) {
                var ws = XLSX.utils.json_to_sheet(data);
                var wb = XLSX.utils.book_new();
                XLSX.utils.book_append_sheet(wb, ws, 'Report');
                XLSX.writeFile(wb, filename + '_' + new Date().toISOString().split('T')[0] + '.xlsx');
                showNotification('Excel report generated successfully', 'success');
            } else if (format === 'csv' && data.length > 0) {
                var csvContent = 'data:text/csv;charset=utf-8,';
                if (data.length > 0) {
                    var headers = Object.keys(data[0]);
                    csvContent += headers.join(',') + '\n';
                    for (var i = 0; i < data.length; i++) {
                        var row = [];
                        for (var j = 0; j < headers.length; j++) {
                            row.push('"' + (data[i][headers[j]] || '') + '"');
                        }
                        csvContent += row.join(',') + '\n';
                    }
                }
                var encodedUri = encodeURI(csvContent);
                var link = document.createElement('a');
                link.setAttribute('href', encodedUri);
                link.setAttribute('download', filename + '_' + new Date().toISOString().split('T')[0] + '.csv');
                document.body.appendChild(link);
                link.click();
                document.body.removeChild(link);
                showNotification('CSV report generated successfully', 'success');
            } else if (format === 'pdf' && data.length > 0) {
                var doc = new jsPDF.jsPDF();
                doc.setFontSize(20);
                doc.text('NEOM NRC9 Report', 20, 20);
                doc.setFontSize(12);
                doc.text('Generated on: ' + new Date().toLocaleDateString(), 20, 35);
                
                var yPosition = 50;
                if (data.length > 0) {
                    var headers = Object.keys(data[0]);
                    doc.setFontSize(8);
                    for (var i = 0; i < Math.min(headers.length, 6); i++) {
                        doc.text(headers[i], 20 + (i * 30), yPosition);
                    }
                    yPosition += 10;
                    
                    for (var i = 0; i < Math.min(data.length, 20); i++) {
                        if (yPosition > 270) break;
                        for (var j = 0; j < Math.min(headers.length, 6); j++) {
                            var text = String(data[i][headers[j]] || '').substring(0, 12);
                            doc.text(text, 20 + (j * 30), yPosition);
                        }
                        yPosition += 8;
                    }
                }
                
                doc.save(filename + '_' + new Date().toISOString().split('T')[0] + '.pdf');
                showNotification('PDF report generated successfully', 'success');
            } else if (data.length === 0) {
                showNotification('No data to export', 'warning');
            }
        }
